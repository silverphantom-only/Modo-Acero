function renderDay(container){ document.getElementById("title").textContent = selectedDate; const events = tasksByDate[selectedDate] || []; for(let h=7; h<=22; h++){ const hour = String(h).padStart(2,'0')+":00"; const row = document.createElement("div"); row.className = "hour-row"; const time = document.createElement("div"); time.className = "hour-time"; time.textContent = hour; const content = document.createElement("div"); content.style.flex = 1; content.style.position = "relative"; // buscar evento que cubra esta hora const event = events.find(e => hour >= e.start && hour < e.end); if(event){ const block = document.createElement("div"); block.style.background = getColor(event.color); block.style.padding = "6px"; block.style.borderRadius = "6px"; block.style.marginBottom = "4px"; block.textContent = `${event.title} (${event.start}-${event.end})`; content.appendChild(block); } // CLICK PARA CREAR BLOQUE row.onclick = ()=>{ const title = prompt("Nombre del bloque"); if(!title) return; const start = hour; const end = prompt("Hora fin (ej: 14:00)"); const color = prompt("Color: blue / purple / black"); if(!tasksByDate[selectedDate]){ tasksByDate[selectedDate] = []; } tasksByDate[selectedDate].push({ start, end, title, color }); save(); render(); }; row.appendChild(time); row.appendChild(content); container.appendChild(row); } }
